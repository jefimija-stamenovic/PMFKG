 7. Kako biste samo pomocu instrukcije testset() ili exchange() izveli medjusobno iskljucenje?
  - Kod:

    #define N 16
    int bolt;

    // testset - verzija
    void P(int i)
    {
        while(true)
        {
            while(!testset()); //Busy Wait
            /* kriticna sekcija */
            bolt = 0;
            /* ostatak koda */
        }
    }

    // exchange - verzija
    void P(int i)
    {
        while(true)
        {
            int key = 1;
            do exchange(bolt, key) while(key != 0); //Busy Wait
            /* kriticna sekcija */
            bolt = 0;
            /* ostatak koda */
        }
    }

    int main()
    {
        bolt = 0;
        parbegin(P, N);
    }
